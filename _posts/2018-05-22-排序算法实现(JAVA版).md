主要介绍了冒泡、选择、插入、归并、快速、堆排等六种排序算法。
稳定性及复杂度。
在综合排序中，一般是这样的，当是基本数据类型时，会使用快速排序；当是自己定义的引用类型，会使用归并排序；当数组长度小于60时使用插入排序。

| 排序算法 | 时间复杂度 | 空间复杂度 | 稳定性 |
| :-----: | :-------: | :------:  | :---: |
| 冒泡排序 |  O($$n^2$$)   | O(1)      | 稳定  |
| 选择排序 |  O($$n^2$$)   | O(1)      | 不稳定  |
| 插入排序 |  O($$n^2$$)   | O(1)      | 稳定  |
| 归并排序 |  O($$nlogn$$) | O(n)      | 稳定  |
| 快速排序 |  O($$nlogn$$) | O(logn)   | 不稳定  |
| 堆排序   |  O($$nlogn$$) | O(1)      | 不稳定  |

package suanfa;

import java.util.Arrays;

public class SortCollection {

    public static void main(String[] args) {
        int[] array = generateRandomArray(34,123);
        int[] arr2 = copyArray(array);
        //bubbleSort(array);
        //rightSortByJdk(arr2);
//        insertSort(array);
        rightSortByJdk(arr2);
        mergeSort(array);
        if (isEqual(array,arr2))
            System.out.println("相等。");
        else
            System.out.println("不相等。");
    }

    /**
     * 冒泡排序
     *
     * @param arr
     */
    public static void bubbleSort(int[] arr) {
        if (arr == null || arr.length < 2) {
            return;
        }
        for (int end = arr.length - 1; end > 0; end--) {
            for (int i = 0; i < end; i++) {
                if (arr[i] > arr[i + 1]) {
                    swap(arr, i, i + 1);
                }
            }
        }
    }


    /**
     * 选择排序
     *
     * @param arr
     */
    public static void selectSort(int[] arr) {
        if (arr == null || arr.length < 2) {
            return;
        }
        for (int i = 0; i < arr.length; i++) {
            for (int j = i + 1; j < arr.length; j++) {
                if (arr[j] < arr[i]) {
                    swap(arr, j, i);
                }
            }
        }
    }

    /**
     * 插入排序
     * 最坏复杂度O（n^2）,最好复杂度O(n)
     * @param arr
     */
    public static  void  insertSort(int[] arr){
        if (arr == null || arr.length < 2) {
            return;
        }
        for(int i = 1; i < arr.length; i++){
            for (int j = i-1; j >=0 && arr[j] > arr[j+1]; j--){
                swap(arr, j,j+1);
            }
        }
    }

    /**
     * 归并排序
     * @param arr
     */
    public  static  void  mergeSort(int[] arr){
        if(arr == null || arr.length < 2)
            return;
        sortProcess(arr, 0 , arr.length-1);
    }

    /**
     * 归并排序，递归进行排序
     * @param arr
     * @param low
     * @param high
     */
    public  static  void  sortProcess(int[] arr, int low, int high){
        if (low == high)
            return;
        int mid = low + ((high - low) >> 1);
        sortProcess(arr, low, mid);
        sortProcess(arr, mid+1, high);
        merge(arr,low, mid, high);
    }

    /**
     * 归并排序中，用于将左右两边整合以有序的方式放到数组中
     * @param arr
     * @param low
     * @param mid
     * @param high
     */
    public static  void merge(int[] arr, int low, int mid, int high){
        int[] mergeArr = new int[high-low+1];
        int left = low;
        int right = mid + 1;

        int i = 0;
        while (left <= mid && right <= high){
            mergeArr[i++] = arr[left] < arr[right] ? arr[left++] : arr[right++];
        }
        while (left <= mid)
            mergeArr[i++] = arr[left++];
        while (right <= high)
            mergeArr[i++] = arr[right++];
        for( i = 0 ; i < mergeArr.length; i++){
            arr[low + i ] = mergeArr[i];
        }
    }

	/**
     * 快速排序-基础版
     *
     * @param arr
     */
    public static void quickSort(int[] arr) {
       if (arr == null || arr.length <2)
           return;
        quickSortProcess(arr, 0, arr.length-1);
    }

    /**
     * 快速排序-基础版
     * @param arr
     * @param low
     * @param high
     */
    public  static  void quickSortProcess(int[] arr, int low, int high){
        if (high <= low)
            return;
        int j = partion(arr, low, high);
        quickSortProcess(arr, low, j - 1);
        quickSortProcess(arr, j + 1, high);
    }

    /**
     * 快速排序-基础版，寻找基准
     * @param arr
     * @param low
     * @param high
     * @return
     */
    public static int partion(int[] arr, int low, int high) {
        int pivot = arr[low];

        while (low < high && arr[high] > pivot) {
            high--;
        }
        //此时high小于pivot，赋值给low，high位置空出来了
        arr[low] = arr[high];
        while (low < high && arr[low] < pivot) {
            low++;
        }
        //此时low位置大于pivot，赋值给high，low位置空出来了
        arr[high] = arr[low];

        arr[low] = pivot;
        return low;

    }


    /**
     * 交换数组中的两个位置的数
     *
     * @param arr
     * @param i
     * @param j
     */
    public static void swap(int[] arr, int i, int j) {
        int temp = arr[j];
        arr[j] = arr[i];
        arr[i] = temp;
    }

    /**
     * 生成随机数组
     *
     * @param size  指定想要的数组长度最大值
     * @param value 数的最大值
     * @return
     */
    public static int[] generateRandomArray(int size, int value) {
        int[] arr = new int[(int) ((size + 1) * Math.random())];
        for (int i = 0; i < arr.length; i++) {
            arr[i] = (int) ((value + 1) * Math.random()) - (int) (value * Math.random());
        }
        return arr;
    }

    /**
     * 利用原生的jdk进行排序，始终正确
     *
     * @param arr
     */
    public static void rightSortByJdk(int[] arr) {
        Arrays.sort(arr);
    }


    /**
     * 用于比较两个数组顺序是否相同
     * @param arr1
     * @param arr2
     * @return
     */
    public static boolean isEqual(int[] arr1, int[] arr2) {

        if ((arr1 == null && arr2 != null) || (arr1 != null && arr2 == null))
            return  false;
        if (arr1 == null && arr2 == null)
            return  true;
        if (arr1.length != arr2.length)
            return  false;
        for(int i = 0; i < arr1.length; i++){
            if (arr1[i] != arr2[i])
                return  false;
        }
        return true;
    }

    public  static  int[] copyArray(int[] arr1){
        return Arrays.copyOf(arr1,arr1.length);
    }
}
